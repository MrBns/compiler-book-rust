# Generating TypeScript

Time to write the code generator! Open `src/codegen.rs`.

## The CodeGen Struct

```rust
// codegen.rs — Walks the AST and generates TypeScript code

use crate::ast::{BinOp, Expr, Stmt, Program};

pub struct CodeGen {
    output: String,      // we append TypeScript code here
    indent_level: usize, // 0 = top level, 1 = inside block, etc.
}

impl CodeGen {
    pub fn new() -> Self {
        CodeGen {
            output: String::new(),
            indent_level: 0,
        }
    }

    // indent() — Return the correct spaces for current indent level
    fn indent(&self) -> String {
        "    ".repeat(self.indent_level) // 4 spaces per level
    }

    // emit() — Append a line of code with proper indentation
    fn emit(&mut self, line: &str) {
        self.output.push_str(&self.indent());
        self.output.push_str(line);
        self.output.push('\n');
    }
}
```

## Generating Statements

```rust
impl CodeGen {
    // gen_program() — Generate TypeScript for the whole program
    pub fn gen_program(&mut self, program: &Program) -> String {
        self.emit("// Generated by Pico Compiler");
        self.emit("// Do not edit this file by hand.");
        self.emit("");

        for stmt in program {
            self.gen_stmt(stmt);
        }

        self.output.clone()
    }

    // gen_stmt() — Generate TypeScript for one statement
    fn gen_stmt(&mut self, stmt: &Stmt) {
        match stmt {
            // let x = value;  →  const x = value;
            Stmt::Let { name, value } => {
                let value_ts = self.gen_expr(value);
                self.emit(&format!("const {} = {};", name, value_ts));
            }

            // return value;  →  return value;
            Stmt::Return(expr) => {
                let expr_ts = self.gen_expr(expr);
                self.emit(&format!("return {};", expr_ts));
            }

            // if condition { ... } else { ... }
            Stmt::If { condition, then_block, else_block } => {
                let cond_ts = self.gen_expr(condition);
                self.emit(&format!("if ({}) {{", cond_ts));

                // Generate the 'then' block with increased indentation
                self.indent_level += 1;
                for s in then_block {
                    self.gen_stmt(s);
                }
                self.indent_level -= 1;

                if let Some(else_stmts) = else_block {
                    self.emit("} else {");
                    self.indent_level += 1;
                    for s in else_stmts {
                        self.gen_stmt(s);
                    }
                    self.indent_level -= 1;
                    self.emit("}");
                } else {
                    self.emit("}");
                }
            }

            // fn add(a, b) { ... }
            //  →  function add(a: any, b: any): any { ... }
            Stmt::Function { name, params, body } => {
                // Build the parameters string: "a: any, b: any"
                let params_ts: Vec<String> = params
                    .iter()
                    .map(|p| format!("{}: any", p))
                    .collect();
                let params_str = params_ts.join(", ");

                self.emit(&format!("function {}({}): any {{", name, params_str));

                self.indent_level += 1;
                for s in body {
                    self.gen_stmt(s);
                }
                self.indent_level -= 1;

                self.emit("}");
                self.emit(""); // empty line after function
            }

            // print(expr);  →  console.log(expr);
            Stmt::Print(expr) => {
                let expr_ts = self.gen_expr(expr);
                self.emit(&format!("console.log({});", expr_ts));
            }

            // expr;  →  expr;
            Stmt::ExprStmt(expr) => {
                let expr_ts = self.gen_expr(expr);
                self.emit(&format!("{};", expr_ts));
            }
        }
    }
}
```

## Generating Expressions

```rust
impl CodeGen {
    // gen_expr() — Generate TypeScript for one expression
    // Returns the expression as a String
    fn gen_expr(&mut self, expr: &Expr) -> String {
        match expr {
            // Number: avoid ".0" at the end for whole numbers
            Expr::Number(n) => {
                if n.fract() == 0.0 {
                    format!("{}", *n as i64) // "42" not "42.0"
                } else {
                    format!("{}", n)         // "3.14"
                }
            }

            // String: add double quotes
            Expr::Str(s) => {
                let escaped = s.replace('"', "\\\"");
                format!("\"{}\"", escaped)
            }

            // Boolean
            Expr::Bool(b) => b.to_string(),

            // Identifier: just the name
            Expr::Ident(name) => name.clone(),

            // Binary expression: (left op right)
            Expr::Binary { left, op, right } => {
                let left_ts  = self.gen_expr(left);
                let right_ts = self.gen_expr(right);
                let op_ts    = self.gen_binop(op);
                // Wrap in parentheses to be safe with precedence
                format!("({} {} {})", left_ts, op_ts, right_ts)
            }

            // Function call: name(arg1, arg2, ...)
            Expr::Call { name, args } => {
                let args_ts: Vec<String> = args
                    .iter()
                    .map(|a| self.gen_expr(a))
                    .collect();
                format!("{}({})", name, args_ts.join(", "))
            }
        }
    }

    // gen_binop() — Convert a BinOp to a TypeScript operator string
    fn gen_binop(&self, op: &BinOp) -> &'static str {
        match op {
            BinOp::Add => "+",
            BinOp::Sub => "-",
            BinOp::Mul => "*",
            BinOp::Div => "/",
            BinOp::Eq  => "===", // strict equality in TypeScript!
            BinOp::Ne  => "!==", // strict not-equal in TypeScript!
            BinOp::Lt  => "<",
            BinOp::Gt  => ">",
            BinOp::Le  => "<=",
            BinOp::Ge  => ">=",
        }
    }
}
```

> **Rust note:** `'static str` means a string that lives for the entire program. String literals like `"+"` are always `'static`.

## The generate Function

Add a public helper function at the bottom of `codegen.rs`:

```rust
// generate() — Public function: take a program, return TypeScript as a String
pub fn generate(program: &Program) -> String {
    let mut codegen = CodeGen::new();
    codegen.gen_program(program)
}
```

## Testing the Code Generator

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use crate::lexer::tokenize;
    use crate::parser::Parser;

    fn compile(src: &str) -> String {
        let tokens = tokenize(src);
        let mut parser = Parser::new(tokens);
        let program = parser.parse();
        generate(&program)
    }

    #[test]
    fn test_let_generates_const() {
        let ts = compile("let x = 42;");
        assert!(ts.contains("const x = 42;"));
    }

    #[test]
    fn test_print_generates_console_log() {
        let ts = compile("print(42);");
        assert!(ts.contains("console.log(42);"));
    }

    #[test]
    fn test_function() {
        let ts = compile("fn double(n) { return n * 2; }");
        assert!(ts.contains("function double(n: any): any {"));
    }

    #[test]
    fn test_if_else() {
        let ts = compile("let x = 1; if x > 0 { print(x); } else { print(x); }");
        assert!(ts.contains("if ((x > 0)) {"));
        assert!(ts.contains("} else {"));
    }
}
```

Run:

```bash
cargo test
```

## Running a Full Example

Create a test file `example.pico`:

```
fn greet(name) {
    print(name);
}

let x = 10;
let y = 20;
let sum = x + y;

if sum > 25 {
    greet("Big number!");
} else {
    greet("Small number.");
}
```

Expected TypeScript output:

```typescript
// Generated by Pico Compiler
// Do not edit this file by hand.

function greet(name: any): any {
    console.log(name);
}

const x = 10;
const y = 20;
const sum = (x + y);
if ((sum > 25)) {
    greet("Big number!");
} else {
    greet("Small number.");
}
```
